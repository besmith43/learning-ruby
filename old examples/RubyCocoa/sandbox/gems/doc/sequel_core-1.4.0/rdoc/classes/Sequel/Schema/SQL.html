58:         if column[:type] == :check
59:           return constraint_definition_sql(column)
60:         end
61:         sql = "#{literal(column[:name].to_sym)} #{type_literal(TYPES[column[:type]])}"
62:         column[:size] ||= 255 if column[:type] == :varchar
63:         elements = column[:size] || column[:elements]
64:         sql << "(#{literal(elements)})" if elements
65:         sql << UNSIGNED if column[:unsigned]
66:         sql << UNIQUE if column[:unique]
67:         sql << NOT_NULL if column[:null] == false
68:         sql << NULL if column[:null] == true
69:         sql << " DEFAULT #{literal(column[:default])}" if column.include?(:default)
70:         sql << PRIMARY_KEY if column[:primary_key]
71:         sql << " #{auto_increment_sql}" if column[:auto_increment]

74:           sql << "(#{column[:key]})" if column[:key]
