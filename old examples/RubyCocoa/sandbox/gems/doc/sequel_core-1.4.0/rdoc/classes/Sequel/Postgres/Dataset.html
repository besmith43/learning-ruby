496:       def compile_converter(columns, translators)
497:         used_columns = []
498:         kvs = []
499:         columns.each_with_index do |column, idx|
500:           next if used_columns.include?(column)
501:           used_columns << column
502:         
503:           if !AUTO_TRANSLATE and translator = translators[idx]
504:             kvs << ":\"#{column}\" => ((t = r[#{idx}]) ? t.#{translator} : nil)"
505:           else
506:             kvs << ":\"#{column}\" => r[#{idx}]"
507:           end
508:         end
509:         eval("lambda {|r| {#{kvs.join(COMMA_SEPARATOR)}}}")
510:       end
