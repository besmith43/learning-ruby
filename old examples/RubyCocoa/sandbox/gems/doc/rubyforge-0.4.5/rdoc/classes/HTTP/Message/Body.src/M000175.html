    def dump(dev = '')
      if @body.respond_to?(:read)
        begin
          while true
            chunk = @body.read(@chunk_size)
            break if chunk.nil?
            dev << dump_chunk(chunk)
          end
        rescue EOFError
        end
        dev << (dump_last_chunk + CRLF)
      else
        dev << @body
      end
      dev
    end
