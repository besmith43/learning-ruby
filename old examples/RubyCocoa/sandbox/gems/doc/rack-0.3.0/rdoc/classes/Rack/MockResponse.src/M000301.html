    def initialize(status, headers, body, errors=StringIO.new(""))
      @status = status.to_i

      @original_headers = headers
      @headers = Rack::Utils::HeaderHash.new
      headers.each { |field, values|
        values.each { |value|
          @headers[field] = value
        }
      }

      @body = ""
      body.each { |part| @body << part }

      @errors = errors.string
    end
