    def initialize(map)
      @mapping = map.map { |location, app|
        if location =~ %r{\Ahttps?://(.*?)(/.*)}
          host, location = $1, $2
        else
          host = nil
        end

        location = ""  if location == "/"

        [host, location, app]
      }.sort_by { |(h, l, a)| -l.size } # Longest path first
    end
