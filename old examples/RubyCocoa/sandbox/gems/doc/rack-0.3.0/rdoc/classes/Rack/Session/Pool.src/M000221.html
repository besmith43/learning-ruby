      def initialize(app, options={})
        @app = app
        @key = options[:key] || "rack.session"
        @default_options = {:domain => nil,
          :path => "/",
          :expire_after => nil}.merge(options)
        @pool = Hash.new
        @default_context = context app, &nil
      end
