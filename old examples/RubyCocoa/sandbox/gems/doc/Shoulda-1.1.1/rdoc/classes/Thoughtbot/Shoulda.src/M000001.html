    def should(name, &blk)
      if Shoulda.current_context
        Shoulda.current_context.should(name, &blk)
      else
        context_name = self.name.gsub(/Test/, "")
        context = Thoughtbot::Shoulda::Context.new(context_name, self) do
          should(name, &blk)
        end
        context.build
      end
    end
