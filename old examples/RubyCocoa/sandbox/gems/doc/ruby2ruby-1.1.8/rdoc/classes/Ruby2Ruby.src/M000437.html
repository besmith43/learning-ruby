  def process_argscat(exp)
    args = []

    ary = exp.shift
    ary.shift # :array
    until ary.empty? do
      args << process(ary.shift)
    end

    args << "*#{process(exp.shift)}"
    args << process(exp.shift) unless exp.empty? # optional block arg

    args.join ', '
  end
