  def process_dasgn_curr(exp)
    lhs = exp.shift.to_s
    rhs = (exp.empty? ? nil : exp.shift)
    if rhs.nil? then
      if self.context[1] == :block then
        return ''
      end

      return lhs
    end
    return "#{lhs} = #{process rhs}" unless rhs.first == :dasgn_curr

    # keep recursing. ensure that the leaf node assigns to _something_
    "#{lhs} = #{process rhs}"
  end
